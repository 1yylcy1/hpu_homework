 #include<iostream>
#include <string.h>
using namespace std;
typedef struct BiTNode
{
	char data;
	struct BiTNode *lchild,*rchild;
}BiTNode,*BiTree;
void CreateBiTree(BiTree &T,char S[],int &i)
{//先序建立二叉树
	if(S[i] == '0'){
        T=NULL;
    }else{
        T=new BiTNode;
        T->data=S[i];
        CreateBiTree(T->lchild,S,++i);
        CreateBiTree(T->rchild,S,++i);
    }
}
int Depth(BiTree T)
{//二叉树高度的计算
	int m,n;
	if(!T){
        return 0;
    }else{
        m = Depth(T->lchild);
        n = Depth(T->rchild);
        if(m>n) return m+1;
        else return n+1;
    }
}
